        async function processFilesWithProgress(files) {
            return new Promise((resolve, reject) => {
                const results = [];
                const total = files.length;
                let processed = 0;
                let failed = 0;
                
                const processNextBatch = (index) => {
                    if (index >= total) {
                        if (processed === 0 && failed > 0) {
                            reject(new Error("æ— æ³•å¤„ç†ä»»ä½•éŸ³ä¹æ–‡ä»¶"));
                        } else {
                            resolve(results);
                        }
                        return;
                    }
                    
                    const batchSize = 5;
                    const endIndex = Math.min(index + batchSize, total);
                    
                    for (let i = index; i < endIndex; i++) {
                        const file = files[i];
                        
                        if (currentFileURLs.has(file.name)) {
                            console.warn(`è·³è¿‡é‡å¤æ–‡ä»¶: ${file.name}`);
                            processed++;
                            continue;
                        }
                        
                        try {
                            const fileURL = URL.createObjectURL(file);
                            currentFileURLs.add(file.name);
                            
                            results.push({
                                index: results.length + 1,
                                name: file.name.replace(/\.[^/.]+$/, ""),
                                path: fileURL,
                                file: file
                            });
                            processed++;
                        } catch (error) {
                            console.error(`å¤„ç†æ–‡ä»¶ ${file.name} å¤±è´¥:`, error);
                            failed++;
                        }
                    }
                    
                    const percent = Math.round((processed / total) * 100);
                    uploadProgressBar.style.width = `${percent}%`;
                    progressText.textContent = `${processed}/${total} æ–‡ä»¶å·²å¤„ç†${failed > 0 ? ` (${failed} ä¸ªå¤±è´¥)` : ''}`;
                    
                    setTimeout(() => processNextBatch(endIndex), 10);
                };
                
                processNextBatch(0);
            });
        }
